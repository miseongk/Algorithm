# DFS λ©”μ„λ“ μ •μ
def dfs(graph, v, visited):
    # (1) ν„μ¬ λ…Έλ“λ¥Ό λ°©λ¬Έ μ²λ¦¬
    visited[v] = True
    print(v, end=' ')

    # (2) ν„μ¬ λ…Έλ“μ™€ μ—°κ²°λ λ‹¤λ¥Έ λ…Έλ“λ¥Ό μ¬κ·€μ μΌλ΅ λ°©λ¬Έ
    for i in graph[v]:
        if not visited[i]:
            dfs(graph, i, visited)

# DFSλ΅ νΉμ • λ…Έλ“λ¥Ό λ°©λ¬Έν•κ³  μ—°κ²°λ λ¨λ“  λ…Έλ“λ“¤λ„ λ°©λ¬Έ
def dfs(x, y):
    # μ£Όμ–΄μ§„ λ²”μ„λ¥Ό λ²—μ–΄λ‚λ” κ²½μ°μ—λ” μ¦‰μ‹ μΆ…λ£
    if x <= -1 or x >= n or y <= -1 or y >= m:
        return False

    # ν„μ¬ λ…Έλ“λ¥Ό μ•„μ§ λ°©λ¬Έν•μ§€ μ•μ•λ‹¤λ©΄
    if graph[x][y] == 0:
        # ν•΄λ‹Ή λ…Έλ“λ¥Ό λ°©λ¬Έ μ²λ¦¬
        graph[x][y] = 1

        # μƒ, ν•, μΆ, μ°μ μ„μΉλ“¤λ„ λ¨λ‘ μ¬κ·€μ μΌλ΅ νΈμ¶
        dfs(x-1, y)
        dfs(x, y-1)
        dfs(x+1, y)
        dfs(x, y+1)
        return True
    return False

n, m = map(int, input().split())

# 2μ°¨μ› λ¦¬μ¤νΈμ λ§µ μ •λ³΄ μ…λ ¥ λ°›κΈ°

# λ¨λ“  λ…Έλ“(μ„μΉ)μ— λ€ν•μ—¬ μλ£μ μ±„μ°κΈ°

    for j in range(m):
        if dfs(i, j) == True:
            result += 1
